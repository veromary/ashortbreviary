\latin{Retríbue servo tuo, vivífica me: * et custódiam sermónes tuos:}
\vern{(Ghimel) Give bountifully to thy servant, enliven me * and I shall keep thy words.}
\latin{Revéla óculos meos: * et considerábo mirabília de lege tua.}
\vern{Open thou my eyes: * and I will consider the wondrous things of thy law.}
\latin{Íncola ego sum in terra: * non abscóndas a me mandáta tua.}
\vern{I am a sojourner on the earth: * hide not thy commandments from me.}
\latin{Concupívit ánima mea desideráre iustificatiónes tuas, * in omni témpore.}
\vern{My soul hath coveted to long for thy justifications, * at all times.}
\latin{Increpásti supérbos: * maledícti qui declínant a mandátis tuis.}
\vern{Thou hast rebuked the proud: * they are cursed who decline from thy commandments.}
\latin{Aufer a me oppróbrium, et contémptum: * quia testimónia tua exquisívi.}
\vern{Remove from me reproach and contempt: * because I have sought after thy testimonies.}
\latin{Étenim sedérunt príncipes, et advérsum me loquebántur: * servus autem tuus exercebátur in iustificatiónibus tuis.}
\vern{For princes sat, and spoke against me: * but thy servant was employed in thy justifications.}
\latin{Nam et testimónia tua meditátio mea est: * et consílium meum iustificatiónes tuæ.}
\vern{For thy testimonies are my meditation: * and thy justifications my counsel.}
\latin{Adhæsit paviménto ánima mea: * vivífica me secúndum verbum tuum.}
\vern{(Daleth) My soul hath cleaved to the pavement: * quicken thou me according to thy word.}
\latin{Vias meas enuntiávi, et exaudísti me: * doce me iustificatiónes tuas.}
\vern{I have declared my ways, and thou hast heard me: * teach me thy justifications.}
\latin{Viam iustificatiónum tuárum ínstrue me: * et exercébor in mirabílibus tuis.}
\vern{Make me to understand the way of thy justifications: * and I shall be exercised in thy wondrous works.}
\latin{Dormitávit ánima mea præ tædio: * confírma me in verbis tuis.}
\vern{My soul hath slumbered through heaviness: * strengthen thou me in thy words.}
\latin{Viam iniquitátis ámove a me: * et de lege tua miserére mei.}
\vern{Remove from me the way of iniquity: * and out of thy law have mercy on me.}
\latin{Viam veritátis elégi: * iudícia tua non sum oblítus.}
\vern{I have chosen the way of truth: * thy judgments I have not forgotten.}
\latin{Adhæsi testimóniis tuis, Dómine: * noli me confúndere.}
\vern{I have stuck to thy testimonies, O Lord: * put me not to shame.}
\latin{Viam mandatórum tuórum cucúrri, * cum dilatásti cor meum.}
\vern{I have run the way of thy commandments, * when thou didst enlarge my heart.}

\setcounter{versecount}{0}


\rubrics{On Maundy Thursday, all then kneel and recite together the following: }

\gregorioscore{Canticle/an--christus_factus_est--solesmes}

\latin{Christus factus est pro nobis obédiens usque ad mortem.}
\vern{Christ was made for us obedient unto death.}

\rubrics{On Good Friday is added:}

\gregorioscore{Canticle/Christusfactus2}

\latin{mortem autem crucis.}
\vern{death on the cross.}

\rubrics{On Holy Saturday is added:}

\gregorioscore{Canticle/Christusfactus3}

\latin{propter quod et Deus exaltavit illum, et dedit illi nomen, quod est super omne nomen.}
\vern{therefore God has exalted him, and given him a name that is above all names.}

\rubrics{All say the Pater Noster silently}

\rubrics{The Collect is said without Oremus %p.~\pageref{collect} 
again dropping the voice on the last syllable as in the psalms.}

\latin{\lettrine{R}{éspice,} quǽsumus, Dómine, super hanc famíliam tuam, pro qua Dóminus noster Iesus Christus non dubitávit mánibus tradi nocéntium, et crucis subíre torméntum:}
\vern{Look down, we beseech thee, O Lord, on this thy family, for which our Lord Jesus Christ did not hesitate to be delivered up into the hands of wicked men, and to suffer the torment of the Cross.}

\rubrics{finishing silently with:}

\latin{Qui tecum vivit et regnat in unitáte Spíritus Sancti Deus per ómnia sǽcula sæculórum.
R. Amen.}
\vern{Who with thee liveth and reigneth, in the unity of the Holy Spirit, one God, world without end. Amen.
R. Amen.}
